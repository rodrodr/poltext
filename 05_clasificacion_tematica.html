6.3 Extracción de temas

Código
library(tenet)
# Filtra el archivo de intervenciones para seleccionar:
# 1) la Ley de la Libertad Sexual
# 2) solo el debate entre diputados
# 3) excluye la moderación y solo mantiene el posicionamiento
#    de los distintos partidos
si <- spa.sessions

si <- si[grep("121/000062",si$issue.details),]

si <- si[si$session.number==124 & 
           si$issue.type=="Debate" &
           si$rep.condition!="Miembro de la mesa",]


# Agrega las intervenciones por partido político
si <- aggregate(list(text=si$speech.text), 
                by=list(partido=si$rep.party),
                paste0,
                collapse="\n")

# Crea el corpus
ci <- corpus(si)
docnames(ci) <- si$partido



pcaScatter( ci, palette=pal$cat.brewer.Dark2.8[1:4],
            title = "Ley del Sí es Sí",            
            min.freq = 5)
